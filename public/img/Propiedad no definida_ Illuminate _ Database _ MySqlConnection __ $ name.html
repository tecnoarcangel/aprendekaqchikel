                                  public function handle()
{
    $users = DB::table('users')
        ->select('id')
        ->where('db_name', null)
        ->limit(10)->get();

    foreach ($users as $user) {

        $dbName = $user->databaseName;

        $db = DB::statement("CREATE DATABASE IF NOT EXISTS $dbName");
        if ($db !== null) {
            $user->update(['db_name' => $dbName]);
        }
    }
}
